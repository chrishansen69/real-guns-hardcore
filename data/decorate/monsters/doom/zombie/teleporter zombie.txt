ACTOR TeleporterZombie
{
    Health 40
    Radius 20
    Height 56
    Speed 30
    PainChance 70
    //Translation "192:207=112:127"
    MONSTER 
    +FLOORCLIP
	+MISSILEEVENMORE
    SeeSound "grunt/sight"
    PainSound "grunt/pain"
    DeathSound "grunt/death"
    ActiveSound "grunt/active"
    Obituary "%o got confused by Teleporter Zombie."
    States
    {
    Spawn:
        POSS A 0
		POSS A 0 Thing_SetSpecial(0,226,613,0,random(10,20))
	    POSS A 0 A_GiveInventory("XPAmount", random(10, 20))
	    POSS P 0 ACS_ExecuteAlways(651,0,0,0,0)
        KSWZ AB 10 A_Look        
        goto Spawn+4
    See:
		KSWZ A 0 A_Jump(255, "ChaseBlink","ChaseBlink","ChaseBlink", "WanderBlink")
	ChaseBlink:	    
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
		KSWZ A 0 A_SetTranslucent(0.0)
		KSWZ A 0 A_ChangeFlag("FLOAT", 1)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 1)
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
		KSWZ A 30
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_Chase
		KSWZ A 0 A_ChangeFlag("FLOAT", 0)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 0)
		KSWZ A 0 A_SetTranslucent(1.0)		
		KSWZ A 1 A_Chase
		KSWZ AAAAAAAAAAAAAAAAAAAAA 0 A_Chase
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
        Loop
	WanderBlink:
		KSWZ A 0 A_ChangeFlag("FRIGHTENED", 0)
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
		KSWZ A 0 A_SetTranslucent(0.0)
		KSWZ A 0 A_ChangeFlag("FLOAT", 1)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 1)
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
		KSWZ A 30
		TNT1 AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 0 A_Wander
		KSWZ A 0 A_ChangeFlag("FLOAT", 0)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 0)
		KSWZ A 0 A_SetTranslucent(1.0)
		KSWZ A 1 A_Chase
		KSWZ AAAAAAAAAAAAAAAAAAAAA 0 A_Chase
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
        Loop
    Missile:
		KSWZ A 0 A_SpawnItemEx("TeleportFog")
		KSWZ A 0 A_ChangeFlag("FLOAT", 0)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 0)
		KSWZ A 0 A_SetTranslucent(1.0)
		KSWZ E 10 A_FaceTarget
		KSWZ E 0 A_JumpIfTargetInLOS(1)
		goto ChaseBlink
		KSWZ E 4 bright A_BspiAttack
		KSWZ E 6 A_FaceTarget
		KSWZ E 0 A_JumpIfTargetInLOS(1)
		goto ChaseBlink
		KSWZ E 4 bright A_BspiAttack
		KSWZ E 6 A_FaceTarget
		KSWZ E 0 A_JumpIfTargetInLOS(1)
		goto ChaseBlink
		KSWZ E 4 bright A_BspiAttack
		KSWZ E 6 A_FaceTarget
		KSWZ E 0 A_JumpIfTargetInLOS(1)
		goto ChaseBlink
		KSWZ E 4 bright A_BspiAttack
		KSWZ E 6 A_FaceTarget
		KSWZ E 4 bright A_BspiAttack
		goto WanderBlink
    Pain:
        KSWZ G 2
        KSWZ G 2 A_Pain
		KSWZ A 0 A_ChangeFlag("FRIGHTENED", 1)
        Goto WanderBlink
    Death:
		KSWZ A 0 A_ChangeFlag("FLOAT", 0)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 0)
		KSWZ A 0 A_SetTranslucent(1.0)
        KSWZ H 5
        KSWZ H 0 ACS_ExecuteAlways(648,0,0,0,0)
        KSWZ I 5 A_Scream
        KSWZ J 5 A_Fall
        KSWZ K 5
        KSWZ L 5
        KSWZ M 5
        KSWZ N -1
        Stop
    XDeath:
		KSWZ A 0 A_ChangeFlag("FLOAT", 0)
		KSWZ A 0 A_ChangeFlag("NOGRAVITY", 0)
		KSWZ A 0 A_SetTranslucent(1.0)
        KSWZ O 5
        KSWZ O 0 ACS_ExecuteAlways(648,0,0,0,0)
		KSWZ P 5 A_XScream
        KSWZ Q 5 A_Fall
        KSWZ RSTUV 5
        KSWZ W -1
        Stop
    Raise:
        KSWZ NMLKJIH 5
        Goto See
    }
}
