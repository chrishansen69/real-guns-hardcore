actor DoomImp_2 : RGAMonster
{
  spawnid 5
  obituary "%o was burned by an imp."
  hitobituary "%o was slashed by an imp."
  health 40
  radius 20
  height 56
  mass 100
  speed 8
  painchance 200
  seesound "imp/sight"
  painsound "imp/pain"
  deathsound "imp/death"
  activesound "imp/active"
  MONSTER
  +FLOORCLIP
  +MISSILEMORE
  dropitem "HealthBonus" 196
  dropitem "HealthBonus" 196
  Species "Imp"
  states
  {
  Spawn:
    POSS A 0 
    POSS A 0 Thing_SetSpecial(0,226,613,0,random(6,12))
    POSS A 0 A_GiveInventory("XPAmount", random(6, 12))
    TROO AB 10 A_Look
    goto Spawn+3
  See:
    TROO AABBCCDD 3 A_Chase
    loop
  Melee:
  Missile:
    TROO EF 8 A_FaceTarget
    TROO G 6 A_CustomComboAttack("DoomImpBall",32,5,"imp/melee")
    goto See
  Missile2:
    TROO EF 8 A_FaceTarget
    TROO G 0 A_CustomMissile("DoomImpBall_2",30,0,5)
    TROO G 0 A_CustomMissile("DoomImpBall_2",30,0,-5)
    TROO G 6 A_CustomMissile("DoomImpBall_2",30,0,0)
    goto See
  Pain:
    TROO H 2
    TROO H 2 A_Pain
	TNT1 G 0 A_AlertMonsters
    goto See
  Death:
    TROO I 8
	TROO I 0 A_JumpIfHealthLower(random(-500,-110),"XDeath")
    TROO J 8 A_Scream
    TROO K 6
    TROO L 6 A_NoBlocking
    TROO L 0 ACS_ExecuteAlways(648,0,0,0,0)
    TROO M -1 A_Jump(15,"Res")
    stop
  Res:
    TROO MM 60
	TROO M 0 A_Jump(192, "Res")
    TROO M 5 A_PlaySound("imp/sight")
    TROO M -1 A_SpawnItemEx("DoomImpRes",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION,0)
    stop
  XDeath:
    TROO N 5
    TROO O 5 A_XScream
    TROO P 5
    TROO Q 5 A_NoBlocking
    TROO Q 0 ACS_ExecuteAlways(648,0,0,0,0)
    TROO RST 5
    TROO U -1
    stop
  Raise:
    TROO ML 8
    TROO KJI 6
    goto See
  }
}

actor DoomImpRes : RGAMonster
{
  spawnid 5
  obituary "%o was burned by an imp's ghost."
  hitobituary "%o was slashed by an imp's ghost."
  health 1
  radius 20
  height 56
  mass 100
  speed 10
  painchance 200
  SEESOUND "SpiritImp/Sight"
  DEATHSOUND "SpiritImp/Death"
  ACTIVESOUND "SpiritImp/Active"
  MONSTER
  +FLOORCLIP
  +NOBLOOD
  -SOLID
  Renderstyle Add
  Alpha 0.0
  dropitem "HealthBonus" 1
  Species "Imp"
  states
  {
  Spawn:
    SPIP A 0
    SPIP A 0 A_GiveInventory("XPAmount", random(6, 14))
	SPIP A 0 A_Jump(128, 2)
	SPIP A 0 A_ChangeFlag("SHOOTABLE", 0)
    SPIP A 0 Thing_SetSpecial(0,226,613,0,random(6,14))
    SPIP A 0 A_JumpIfInventory("Awoken",1,"Look")
    SPIP AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA 2 A_FadeIn(0.025)
    Goto Look
  Look:
    SPIP A 0 A_GiveInventory("Awoken",1)
    SPIP AB 10 A_Look
    loop
  See:
	SPIP A 0 A_Jump(140, 3)
	SPIP A 0 A_ChangeFlag("SHOOTABLE", 1)
	SPIP A 0 A_Jump(255, 2)
	SPIP A 0 A_ChangeFlag("SHOOTABLE", 0)	
    SPIP AABBCCDD 3 A_Chase
    loop
  Melee:
  Missile:
    SPIP EF 8 A_FaceTarget
    SPIP G 6 A_CustomMissile("SpiritImpBall",30,0,5)
    goto See
  Missile2:
    SPIP EF 8 A_FaceTarget
    SPIP G 0 A_CustomMissile("SpiritImpBall",30,0,5)
    SPIP G 0 A_CustomMissile("SpiritImpBall",30,0,-5)
    SPIP G 6 A_CustomMissile("SpiritImpBall",30,0,0)
    goto See
  Pain:
    SPIP N 3
    SPIP N 3 A_Pain
	TNT1 N 0 A_AlertMonsters
    Goto See
  Death:
    SPIP PQR 4 
    SPIP S 4 A_SpawnItemEx("HealthBonus",0,0,0,random(-2,2),random(-2,2),random(1,4),0,0,0)
    SPIP T 3 A_Fall
	SPIP T 0 A_KillChildren
	SPIP T 0 ACS_ExecuteAlways(648,0,0,0,0)
    SPIP U 3 
    SPIP V 3 A_NoBlocking
  FadeOut:
    SPIP W 3 A_FadeOut(0.015)
    loop
  Raise:
    SPIP ML 8
    SPIP KJI 6
    goto See
  }
}
