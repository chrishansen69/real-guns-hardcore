ACTOR Demon_2 : RGAMonster
{
  Health 175
  PainChance 180
  Speed 20
  Radius 30
  Height 56
  Mass 400
  Monster
  +FLOORCLIP
  SeeSound "demon/sight"
  AttackSound "demon/melee"
  PainSound "demon/pain"
  DeathSound "demon/death"
  ActiveSound "demon/active"
  Obituary "$OB_DEMONHIT" // "%o was bit by a demon."
  Species "Demon"
  States
  {
  Spawn:
    POSS A 0 
    POSS A 0 Thing_SetSpecial(0,226,613,0,random(8,16))
    POSS A 0 A_GiveInventory("XPAmount", random(8, 16))
    SARG AB 10 A_Look
    goto Spawn+3
  See:
    SARG AABBCCDD 2 A_Chase
    Loop
  Melee:
    SARG EF 8 A_FaceTarget
    SARG G 8 A_SargAttack
    Goto See
  Pain:
    SARG H 2
    SARG H 2 A_Pain
	TNT1 G 0 A_AlertMonsters
    Goto See
  Death:
    SARG I 8
    SARG J 8 A_Scream
    SARG K 4
    SARG L 4 A_NoBlocking
    SARG L 0 ACS_ExecuteAlways(648,0,0,0,0)
    SARG M 4
    SARG N -1
    stop
    //SARG NNNN 115
    //SARG N 15 A_PlaySound("stoner/sight2")
    //SARG N 0 A_SpawnItemEx("StoneDemonRes",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION,0)
    //stop
  Raise:
    SARG N 5
    SARG MLKJI 5
    Goto See
  }
}

actor StoneDemonRes : StoneDemon

    POSS A 0 Thing_SetSpecial(0,226,613,0,random(12,24))
    STND A 0 A_JumpIfInventory("Awoken",1,"Look")
    STND NMLKJI 5
    Goto Look
  Look:
    STND A 0 A_GiveInventory("Awoken",1)
    STND AB 10 A_Look
    loop
  }
